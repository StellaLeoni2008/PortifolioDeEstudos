Sobre o Projeto

HTML CSS JavaScript

O Sorteio de Amigo Secreto foi desenvolvido como parte das atividades do curso Alura – Lógica de Programação: praticando com desafios. O objetivo do projeto foi aplicar conceitos de lógica de programação utilizando JavaScript, criando uma aplicação simples que realiza o sorteio de amigo secreto entre os participantes informados pelo usuário.


Na aplicação, o usuário pode adicionar nomes a uma lista de participantes e, ao realizar o sorteio, o sistema embaralha os nomes e gera automaticamente as combinações de quem irá tirar quem no amigo secreto. O resultado é exibido diretamente na página, mostrando a relação entre os participantes sorteados.

Conceitos Fundamentais Praticados

Manipulação e armazenamento de dados em listas
Adição dinâmica de elementos informados pelo usuário
Atualização progressiva de conteúdo na interface
Implementação de lógica de embaralhamento de dados
Uso de operadores para percorrer e relacionar elementos de uma lista
Geração de relações entre participantes no sorteio
Reinicialização do estado da aplicação para permitir novos sorteios
Organização da lógica do programa para execução passo a passo

Código Fonte

js/app.js 
let amigos = [];

function adicionar() {
  let nome = document.getElementById('nome-amigo').value;
  let listaElement = document.getElementById('lista-amigos');
  if (listaElement.innerHTML == '') {
    listaElement.innerHTML = nome;
  } else {
    listaElement.innerHTML += ', ' + nome;
  }

  document.getElementById('nome-amigo').value = '';
  amigos.push(nome);
}

function sortear() {
  embaralhar(amigos);
  const sorteio = document.getElementById('lista-sorteio');

  for (let i = 0; i < amigos.length; i++) {
    if (i == amigos.length - 1) {
      sorteio.innerHTML = sorteio.innerHTML + amigos[i] + ' -> ' + amigos[0] + '<br>';
    } else {
      sorteio.innerHTML = sorteio.innerHTML + amigos[i] + ' -> ' + amigos[(i + 1) % amigos.length] + '<br>';
    }
  }
}

function embaralhar(listaElement) {
  for (let indice = listaElement.length; indice; indice--) {
    const indiceAleatorio = Math.floor(Math.random() * indice);
    [listaElement[indice - 1], listaElement[indiceAleatorio]] = [listaElement[indiceAleatorio], listaElement[indice - 1]];
  }
}

function reiniciar() {
  nome = document.getElementById('nome-amigo').value = '';
  listaElement = document.getElementById('lista-amigos').innerHTML = '';
  sorteio = document.getElementById('lista-sorteio').innerHTML = '';
  amigos = [];
}
